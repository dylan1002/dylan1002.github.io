标签（空格分隔）： JS

1.运动框架

var timer=null; function startmove() { var oDiv=document.getElementById('div1');

clearInterval(timer); //开始时关闭原有定时器 timer=setInterval(function(){ var speed=1; if(oDiv.offsetLeft>=300) { clearInterval(timer); } else { oDiv.style.left=oDiv.offsetLeft+speed+'px'; } },30);
}
注意：1.开始时把原来的定时器关闭
2.把运动和停止隔开（if/else）

使用例子。
做一个侧边分享框，鼠标悬停滑出，移走滑入。

window.onload=function(){
var oDiv=document.getElementById(‘div1’);
oDiv.onmouseover=function()
{
starmove(0,10);
};
oDiv.onmouseout=function()
{
starmove(-150,-10);
};
}
var timer=null;
function starmove(where,aid){
var oDiv=document.getElementById(‘div1’);

  clearInterval(timer);
  timer=setInterval(function(){
       if(oDiv.offsetLeft==where)
       {
          clearInterval(timer);
       }
       else
       {
        oDiv.style.left=oDiv.offsetLeft+aid+'px';   
        }
      },30);
}
