3.键盘事件

onkeydowm、onkeyup
例子 键盘控制DIV移动

document.onkeydown=function(ev)
{
var oEvent=ev||event;
var oDiv=document.getElementById('div1'); if(oEvent.keyCode==37)
{

oDiv.style.left=oDiv.offsetLeft-10+'px';
} else if(oEvent.keyCode==39) { oDiv.style.left=oDiv.offsetLeft+10+'px'; }
}

5.拖拽

window.onload=function(){ var oDiv=document.getElementById("div1");

var disX=0; var disY=0; oDiv.onmousedown=function(ev) { var oEvent=ev||event; disX=oEvent.clientX-oDiv.offsetLeft; disY=oEvent.clientY-oDiv.offsetTop; document.onmousemove=function(ev) { var oEvent=ev||event;

oDiv.style.left=oEvent.clientX-disX+'px'; oDiv.style.top=oEvent.clientY-disY+'px'; }; document.onmouseup=function(){ document.onmousemove=null; document.onmouseup=null; }; return false; //修复火狐BUG }; };
