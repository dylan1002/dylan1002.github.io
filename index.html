DOM高级操作

标签（空格分隔）： 未分类

表格操作

获取：
tBodies、tHead、tFoot、rows、cells(单元格)

搜索：
1.字符串比较
2.忽略大小写
3.模糊搜索 search
4.多关键字 split

<script>

window.onload=function()
{
oTab=document.getElementById(‘tab1’);
oBtn=document.getElementById(‘btn1’);
oTet=document.getElementById(‘name’);

oBtn.onclick=function()
{ 
   for(var i=0;i<oTab.tBodies[0].rows.length;i++)
   {
     var sTab=oTab.tBodies[0].rows[i].cells[1].innerHTML.toLowerCase();  //大小忽略
     var sTet=oTet.value.toLowerCase();
     var arr=sTet.split(' ');   //多个关键字 用空格隔开

     oTab.tBodies[0].rows[i].style.background='';  
     for(j=0;j<arr.length;j++)
     {
     if(sTab.search(arr[j])!=-1)    //模糊搜索
     {
       oTab.tBodies[0].rows[i].style.background='yellow';
     }    
     }
   }  
}

}
